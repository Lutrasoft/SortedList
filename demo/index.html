Custom sorting

$("#demo02").sortedList({
	sort : [
		// Sort on moved or not DESC
		//	Like you see data, you can use attr or whatever jQuery getter you want
		//	it calls functions like: a.data( "al-moved" ), a.attr( "al-moved" ), etc
		{ desc : { data : "al-moved" } },
		
		// Sort on the class "x"
		function( el ){
			return !$(el).hasClass( "x");
		},
		
		// Sort on the data index ASC
		{ asc : { data : "al-index" } }
	]
});
